Class to store pyramid configuration settings (from the ini file) and provide easier programatic access to them.

Configuration Options

These options can be overridden in the pyramid app ini-file.

# Allow client to specify resource ids.
pyramid_jsonapi.allow_client_ids = False

# Whether or not to add debugging endpoints.
pyramid_jsonapi.debug_endpoints = False

# Whether or not to add debug information to the meta key in returned JSON.
pyramid_jsonapi.debug_meta = False

# Module responsible for populating test data.
pyramid_jsonapi.debug_test_data_module = test_data

# Should /metadata endpoint be enabled?
pyramid_jsonapi.metadata_endpoints = True

# Modules to load to provide metadata endpoints (defaults to all modules in the metadata package).
pyramid_jsonapi.metadata_modules =

# Default pagination limit for collections.
pyramid_jsonapi.paging_default_limit = 10

# Default limit on the number of items returned for collections.
pyramid_jsonapi.paging_max_limit = 100

# Prefix for pyramid route names for view_classes.
pyramid_jsonapi.route_name_prefix = pyramid_jsonapi

# Separator for pyramid route names.
pyramid_jsonapi.route_name_sep = :

# Prefix for api endpoints (if metadata_endpoints is enabled).
pyramid_jsonapi.route_pattern_api_prefix = api

# Prefix for metadata endpoints (if metadata_endpoints is enabled).
pyramid_jsonapi.route_pattern_metadata_prefix = metadata

# "Parent" prefix for all endpoints
pyramid_jsonapi.route_pattern_prefix =

# Separator for pyramid route patterns.
pyramid_jsonapi.route_pattern_sep = /

# File containing jsonschema JSON for validation.
pyramid_jsonapi.schema_file =

# jsonschema schema validation enabled?
pyramid_jsonapi.schema_validation = True
